
class CustomMap {

    constructor(containerId) {
        let container = containerId;
        if (typeof containerId === 'string') {
            container = document.querySelector(containerId);
        }
        if (!container) {
            console.error('Given container not found');
            return;
        }
        const svg = `{{&svgStr}}`;
        container.innerHTML = svg;
        const commonCss = `{{&commonCss}}`;
        const addedCss = `{{&addedCss}}`;
        this.addCssSheet(commonCss, addedCss);
    }

    addCssSheet(...sheets) {
        const styleSheet = document.createElement("style");
        const allStyle = sheets.reduce((styling, cur) => styling + cur, '');
        console.log(allStyle);
        styleSheet.innerText = allStyle;
        document.head.appendChild(styleSheet);
    }
}

new CustomMap('#test');
